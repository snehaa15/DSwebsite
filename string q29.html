#include<stdio.h>
    #include<string.h>
    
    int main() {
        char input[11]; // Assuming a maximum of 10 characters
        int ascii_values[10]; // Array to store ASCII values
        int i, length;
    
        printf("Enter a string (up to 10 characters): ");
        gets(input);
    
        length = strlen(input);
    
        if (length > 10) {
            printf("String is too long. Maximum 10 characters allowed.\n");
            return 1;
        }
    
        // Convert characters to their ASCII values and store them in the array
        for (i = 0; i < length; i++) {
            ascii_values[i] = (int)input[i];
        }
    
        // Display the ASCII values
        printf("ASCII values of characters in the string: ");
        for (i = 0; i < length; i++) {
            printf("%d ", ascii_values[i]);
        }
        printf("\n");
    
        return 0;
    }
