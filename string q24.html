#include<stdio.h>
    #include<stdlib.h>
    #include<string.h>
    
    void main()
    {
        char input[50],subs[15],with[15],dub[15];
        int i=0,j=0,lenght;
        puts("Enter the sentence : ");
        gets(input);
        puts("Enter the word to be replaced");
        gets(subs);
        puts("Enter the word you want to replace with");
        gets(with);
        puts(input);
    
        lenght=strlen(input);
        while (lenght>=0)
        {
            if (input[i]!= ' ')
            // it checks if the current character is not a space (' ').
            // If it's not a space, it appends the character to the dub array and increments j.
            {
                dub[j]=input[i];
                j++;
            }
            //working :- 
            // It then compares the word in dub with the word to be replaced (subs) using strcmp.
            //  If the words match, it prints the replacement word (with), otherwise, it prints the original word. 
            if (input[i]==' '|| input[i]=='\0')
            {
                dub[j]='\0';
                if (strcmp(subs,dub)==0) //compare string
                {
                    printf("%s ",with);
                }
                else
                {
                    printf("%s ",dub);
                }
                j=0;
                //resets j to 0 to start building the next word
            }
            i++;
            //increments i and decrements length to continue processing the next character in the input sentence.
            lenght--;
        }
        
    }
