    int compare(char word[],char word2[])
    {
        int comp = -1;
         int length2=0,length1 = 0, i = 0,incr=0;
        for (i = 0; word[i] != '\0'; i++)
        {
            length1++;
        }
        for (i = 0; word2[i] != '\0'; i++)
        {
            length2++;
        }
        if (length1!=length2)
        {
            return comp;
        }
        for ( i = 0; i < length1; i++)
        {
            if (word[i]==word2[i])
            {
                incr++;
            }
            
        }
        if (incr==length1)
        {
            comp=incr;
            return comp;
        }
        else{
            return -1;
        }
        
    }
