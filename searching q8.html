    void bbl_sort(char string[], int length) {
        int i,j;
        char temp;
        for (i = 0; i < length - 1; i++) {
            for (j = 0; j < length - i - 1; j++) {
                if (string[j] > string[j + 1]) {
                    // Swap the characters if they are in the wrong order
                    temp = string[j];
                    string[j] = string[j + 1];
                    string[j + 1] = temp;
                }
            }
        }
    }
    
    int binarySearch(char string[], char key, int length) {
        int left = 0;
        int right = length - 1;
    
        while (left <= right) {
            int mid = left + (right - left) / 2;
    
            if (string[mid] == key) {
                return mid; // Found the character at position 'mid'
            }
    
            if (string[mid] < key) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
    
        return -1; // Character not found in the string
    }
