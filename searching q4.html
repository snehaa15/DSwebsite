#include<stdio.h>

    int l_srch(char *source, char target) {
        int index = 0; // Initialize index to 0
    
        while (source[index] != '\0') {//end of array
            if (source[index] == target) {
                return index; // Character found at the current index
            }
            index++;
        }
    
        return -1; // Character not found in the string
    }
    
    void main() {
        char input[100]; // Define a character array to store the input string
        char target;
    
        printf("Enter a string: ");
        scanf("%s", input); // Read a string from the user
    
        printf("Enter a character to search: ");
        scanf(" %c", &target); // Read a character to search (note the space before %c)
    
        int index = l_srch(input, target);
    
        if (index != -1) {
            printf("Character '%c' found at index %d in the string.\n", target, index);
        } else {
            printf("Character '%c' not found in the string.\n", target);
        }
    
    }
