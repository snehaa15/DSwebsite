    int b_srch_recur(int ary[], int key,int lb,int up)
    {
        int mid;
        if (lb> up) {
            return -1; // Element not found
        }
            mid=(lb+up)/2;
            if (ary[mid]==key)
            {
                return mid;
            }
            else if (key>ary[mid])
            {
                return b_srch_recur(ary, key, mid + 1, up);
            }
            else
                return b_srch_recur(ary, key, lb, mid - 1); 
    }
    
    int bubble(int ary[], int size)
    {
        int i,j,temp;
        for ( i = size-1; i >0; i--)
        {
            for ( j= 0; j <= i; j++)
            {
                if (ary[j]>ary[j+1])
                {
                    temp=ary[j];
                    ary[j]=ary[j+1];
                    ary[j+1]=temp;
                }
                
            }
            
        }
        return 0;
        
    }
