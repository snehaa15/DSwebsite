#include<stdio.h>
    #include<string.h>
    int convert_to_int(char word[])
    {
        // num = num * 10 + (word[i] - 48); main formula. Explanation : -
        // 1. Why multiply by 10?? -------------------------!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
        // answer is simple. we all are familiar with ones tens hundres places and so on
        // when we convert fisrt we want to place it on first position. hence by *10, 0 remains 0.
        // and the value is added to zero
    
        // where as, on second call, we want to add number on second place not with the first one.
        //  asssme : 1 +5 on second call becomes six. but by 10 multiply, it becomes : 10+5, becomming desired output.
    
        // 2. Why sub. 48?? ---------------------!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
        //  it is becaz the ascii value of 5 is 53. if we do not sub. 48 it will not be the desired output.
        // so, by sub, 48 it becomes the original value but in integer rather then character.
        // 53-48 = 5; ---> desired output
        
        
        int i, num = 0;
        for (i = 0; word[i] != '\0'; i++)
        {
            num = num * 10 + (word[i] - 48);
        }
        return num;
    }
