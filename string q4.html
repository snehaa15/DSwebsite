#include<stdio.h>
    #include<string.h>
    void title(char word[])
    {
        int length = 0, i = 0;
        for (i = 0; word[i] != '\0'; i++)
        {
            length++;
        }
        for (i = 0; i < length; i++)
        {
            if ((word[i] >= 'a' && word[i] <= 'z') || (word[i] >= 'A' && word[i] <= 'Z'))
            {
                if (i == 0 || word[i - 1] == ' ')
                {
                    if (word[i] >= 'a' && word[i] <= 'z')
                    {
                        word[i] = word[i] - 32; // between 97 to 122
                    }
                }
                else
                {
                    if (word[i] >= 'A' && word[i] <= 'Z') // for lower case, if any upper case found besides desired ones.
                    {
                        word[i] = word[i] + 32;
                    }
                }
            }
        }
    }
    void main()
    {
        char Sentence[100];
        puts("Enter the text : ");
        gets(Sentence);
        title(Sentence);
        printf("Sentence after title text convert : %s", Sentence);
    }
