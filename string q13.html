#include<stdio.h>
    #include<string.h>
    void main()
    {
        char Sentence[100], coppied[100];
        int i=0,j=0;
        puts("Enter the text : ");
        gets(Sentence);
         int length = 0;
        for (i = 0; Sentence[i] != '\0'; i++)
        {
            length++;
        }
         for (i = length - 1, j = 0; i >= 0; i--, j++) {
            coppied[j] = Sentence[i];
        }
    
        coppied[j] = '\0';  // Null-terminate the reversed string
    
        printf("Original string: %s\n", Sentence);
        printf("Reversed string: %s\n", coppied);
    }
