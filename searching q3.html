//c program to ENTER A STRING FROM THE USER,
    // ENTER A CHARACTER AND PERFORM LINEAR SEARCH ALGORITHM THROUGH RECURSION.
    
    #include<stdio.h>
    
    int l_srch_recur(char *source, char target, int index) {
        if (*source == '\0') {
            return -1;  // Character not found in the string
        }
        if (*source == target) {
            return index;  // Character found at the current index
        }
        return l_srch_recur(source + 1, target, index + 1);  
        // Recur with the next character and incremented index by 1. 
        //and recur will automatically find the target in ary
    }
    
    int main() {
        char input[100];  // Define a character array to store the input string
        char target;
    
        printf("Enter a string: ");
        gets(input);  // Read a string from the user
        printf("Enter a character to search: ");
        scanf(" %c", &target);  // Read a character to search (note the space before %c)
    
        int index = l_srch_recur(input, target, 0);
    
        if (index != -1) {
            printf("Character '%c' found at index %d in the string.\n", target, index+1);
        } else {
            printf("Character '%c' not found in the string.\n", target);
        }
    
        return 0;
    }
