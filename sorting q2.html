#include<stdio.h>
    #include<time.h>
    
    void selection(int arr[], int n, int start_indx) {
        //check if start index has reached it's end.
        //indicating array is completed
        if (start_indx >= n - 1) {
            return;
        }
        //it will keep track of minimum value of unsorted array
        int min_indx = start_indx;
        //for loop to find minimum value in array
        for (int i = start_indx + 1; i < n; i++) {
            if (arr[i] < arr[min_indx]) {
                min_indx = i;
            }
        }
    //i minindex and startindex differ it indicates
    //a new smaller element was found in unsorted array
        if (min_indx != start_indx) {
            //so let's swap the element and bring it to the first
            int temp = arr[start_indx];
            arr[start_indx] = arr[min_indx];
            arr[min_indx] = temp;
        }
    //recursively calling the function with incremented value of start index
        selection(arr, n, start_indx + 1);
    }
    
    void main() {
        int i,n;
        printf("Enter the number of elements in the array: ");
        scanf("%d", &n);
    
        int arr[n];
        printf("Enter %d integers:\n", n);
        for (i = 0; i < n; i++) {
            scanf("%d", &arr[i]);
        }
    
        // Record the start time
        clock_t start = clock();
    
        // Perform Selection Sort recursively
        selection(arr, n, 0);
    
        // Record the end time
        clock_t end = clock();
    
        // Calculate the time taken in seconds
        double time_complexity = ((double)(end - start)) / CLOCKS_PER_SEC;
    
        printf("Sorted array using Selection Sort:\n");
        for (int i = 0; i < n; i++) {
            printf("%d ", arr[i]);
        }
    
        printf("\nTime taken: %f seconds\n", time_complexity);
    
    }
