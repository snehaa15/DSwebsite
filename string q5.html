#include<stdio.h>
    #include<string.h>
    void toggle(char word[])
    {
        int i = 0;
        for (i = 0; word[i] != '\0'; i++)
        {
            if (word[i] >= 97 && word[i] <= 122)
            {
                word[i] = word[i] - 32;
            }
            else if(word[i] >= 65 && word[i] <= 90)
            {
                word[i] = word[i] + 32;
            }
        }
    }
    void main()
    {
        char Sentence[100];
        puts("Enter the text : ");
        gets(Sentence);
        toggle(Sentence);
        printf("Sentence after title text convert : %s", Sentence);
    }
