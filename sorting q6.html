    void insort_rec(int arr[], int n)
    {
        int i, j, last;
        if (n <= 1)
        {
            return; // 1 element is always sorted
        }
    
        // Sort the first n-1 elements
        insort_rec(arr, n - 1);
    
        // Insert the last element in its correct position
        last = arr[n - 1];
        j = n - 2;
        //The last element of the array is stored in the variable last.
            // This inserts the last element into its correct sorted position within the first
            // n -1 elements of the array.This is done by shifting elements that are 
            //greater than last one position to the right
            //until the correct position is found :
            while (j >= 0 && arr[j] > last)
            {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = last;
    }
