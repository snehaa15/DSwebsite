    char specific(char word[], int find)
    {
        int length = 0, i = 0;
        char ch='1';
        for (i = 0; word[i] != '\0'; i++)
        {
            length++;
        }
        if (find > length)
        {
            return ch;
        }
        ch = word[find-1];
        return ch;
    }
