#include<stdio.h>
    #include<stdlib.h>
    struct q
    {
        int data;
        struct q *next;
    };
        struct q *front = NULL;
        struct q *rear = NULL;
    void enqueue(int value)
    {
        struct q*nn=(struct q*)malloc(sizeof(struct q));
        if (nn==NULL)
        {
            printf("\nqueue is full");
        }
        else
        {
            nn->data=value;
            nn->next=NULL;
            if (front==NULL)
            {
                front=rear=nn;
            }
            else
            {
                rear->next=nn;
                rear=nn;//so that now rear is updated and not pointing to the past value. now pointing updated rear value
            }
            
        }
        
    }
    int dequeue()
    {
        int val=-1;
        struct q*check=front;
        if(front==NULL)
        {
            printf("Queue is empty");
        }
        else
        {
            front=front->next;
            val=check->data;//to know which element is deleted
            free(check);
        }
        return val;
    }
    // print the linked list
    void traverse(struct q *point)
    {
        if (point == NULL)
        {
            printf("NO value found to print. Insert some values first");
        }
        else
        {
            while (point != NULL)
            {
                printf("%d, ", point->data);
                point = point->next;
            }
        }
    }
