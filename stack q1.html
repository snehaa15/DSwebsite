#include<stdio.h>
    int push(int ary[],int MAX,int value, int top)
    {
        if (top<=MAX)
        {
            top++;
        ary[top]=value;
        }
        return top;
    
    }
    int pop(int top)
    {
        if (top==-1)
        {
            return top;
        }
        else
            top--;
        return top;
    }
    int peek(int top)
    {
        if (top==-1)
        {
            return 2;
        }
        return top;
        
    }
    int isEmpty(int top)
    {
        if (top==-1)
        {
            return 0; //true
        }
        return 1;//false
        
    }
    int isFull(int top,int MAX)
    {
            if (top==MAX)
        {
            return 0; //true
        }
        return 1;//false
    }
