#include<stdio.h>

    #include<time.h>
    int search(int ary[],int size,int key)
    {
        int loc=-1,i;
        for ( i = 0; i < size; i++)
        {
            if (ary[i]==key)
            {
                loc=i;
                break;
            }
            
        }
        return loc;
        
    }
    
    void main()
    {
        int ary[10],i,count,loc,num;
        printf("Enter your array size : ");
        scanf("%d",&count);
        for ( i = 0; i < count; i++)
        {
            printf("\nEnter %d value : ",i+1);
            scanf("%d",&ary[i]);
        }
        printf("\nEnter the number you want to search : ");
        scanf("%d",&num);
         clock_t start=clock();
        loc =search(ary,count,num);
        clock_t end=clock();
        double time_complexity=((double)(end-start))/CLOCKS_PER_SEC;
        if (loc==-1)
        {
            printf("\nElement not found in array.");
        }
        else
        {
            printf("Element found at position  : %d",loc+1);
        }
         printf("\nTime complexity is : %f",time_complexity);
    }
