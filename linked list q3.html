    struct dll
    {
        int data;
        struct dll *prev;
        struct dll *next;
    };
    
    struct dll *head = NULL;
    
    //is empty
    int is_empty()
    {
        return head == NULL;
    }
    
    // display  the list
    void display()
    {
        struct dll *current = head;
        if (is_empty())
        {
            printf("The list is empty.\n");
            return;
        }
    
        printf("List elements: ");
        while (current != NULL)
        {
            printf("%d -> ", current->data);
            current = current->next;
        }
        printf("NULL\n");
    }
    
    // insert at beginning
    void insert_at_begin(int value)
    {
        struct dll *newNode = (struct dll *)malloc(sizeof(struct dll));
        newNode->data = value;
        newNode->prev = NULL;
        newNode->next = head;
    
        if (head != NULL)
        {
            head->prev = newNode;
        }
    
        head = newNode;
        printf("%d inserted at the beginning.\n", value);
    }
    
    // at pos
    void insert_at_pos(int value, int position)
    {
        if (position < 1)
        {
            printf("Invalid position.\n");
            return;
        }
    
        struct dll *newNode = (struct dll *)malloc(sizeof(struct dll));
        newNode->data = value;
    
        if (position == 1 || head == NULL)
        {
            newNode->prev = NULL;
            newNode->next = head;
            if (head != NULL)
            {
                head->prev = newNode;
            }
            head = newNode;
            printf("%d inserted at position 1.\n", value);
            return;
        }
    
        struct dll *current = head;
        int count = 1;
        while (count < position - 1 && current->next != NULL)
        {
            current = current->next;
            count++;
        }
    
        newNode->prev = current;
        newNode->next = current->next;
    
        if (current->next != NULL)
        {
            current->next->prev = newNode;
        }
    
        current->next = newNode;
        printf("%d inserted at position %d.\n", value, position);
    }
    
    //insert at end
    void insert_at_end(int value)
    {
        struct dll *newNode = (struct dll *)malloc(sizeof(struct dll));
        newNode->data = value;
        newNode->next = NULL;
    
        if (head == NULL)
        {
            newNode->prev = NULL;
            head = newNode;
            printf("%d inserted at the end.\n", value);
            return;
        }
    
        struct dll *current = head;
        while (current->next != NULL)
        {
            current = current->next;
        }
    
        newNode->prev = current;
        current->next = newNode;
        printf("%d inserted at the end.\n", value);
    }
    
    // del fro, beg
    void del_from_beginn()
    {
        if (is_empty())
        {
            printf("The list is empty. Nothing to delete.\n");
            return;
        }
    
        struct dll *temp = head;
        head = head->next;
    
        if (head != NULL)
        {
            head->prev = NULL;
        }
    
        free(temp);
        printf("Deleted from the beginning.\n");
    }
    
    // delete from end
    void del_from_end()
    {
        if (is_empty())
        {
            printf("The list is empty. Nothing to delete.\n");
            return;
        }
    
        struct dll *current = head;
        while (current->next != NULL)
        {
            current = current->next;
        }
    
        if (current->prev != NULL)
        {
            current->prev->next = NULL;
        }
    
        free(current);
        printf("Deleted from the end.\n");
    }
    
    // Function to delete a node from a specified position
    void del_from_pos(int position)
    {
        if (is_empty())
        {
            printf("The list is empty. Nothing to delete.\n");
            return;
        }
    
        if (position < 1)
        {
            printf("Invalid position.\n");
            return;
        }
    
        if (position == 1)
        {
            del_from_beginn();
            return;
        }
    
        struct dll *current = head;
        int count = 1;
        while (count < position && current->next != NULL)
        {
            current = current->next;
            count++;
        }
    
        if (count < position)
        {
            printf("Position not found in the list.\n");
            return;
        }
    
        if (current->prev != NULL)
        {
            current->prev->next = current->next;
        }
    
        if (current->next != NULL)
        {
            current->next->prev = current->prev;
        }
    
        free(current);
        printf("Deleted from position %d.\n", position);
    }
    
    // Function to search for a value in the list
    int search(int value)
    {
        struct dll *current = head;
        int position = 1;
        while (current != NULL)
        {
            if (current->data == value)
            {
                printf("%d found at position %d.\n", value, position);
                return 1;
            }
            current = current->next;
            position++;
        }
    
        printf("%d not found in the list.\n", value);
        return 0;
    }
